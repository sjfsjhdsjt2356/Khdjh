# --- DATABASE INITIALIZATION ---
def setup_db():
    conn = sqlite3.connect('football_rpg.db')
    c = conn.cursor()
    
    # 1. Clubs Table
    # Stores basic info, training levels, and stadium names
    c.execute('''CREATE TABLE IF NOT EXISTS clubs (
        user_id INTEGER PRIMARY KEY,
        club_name TEXT,
        stadium_name TEXT,
        balance INTEGER DEFAULT 1000000,
        training_lvl INTEGER DEFAULT 3,
        young_training_lvl INTEGER DEFAULT 1
    )''')
    
    # 2. Players Table
    # Stores all 8 stats + age, position, and status (Starter vs Sub)
    c.execute('''CREATE TABLE IF NOT EXISTS players (
        player_id INTEGER PRIMARY KEY AUTOINCREMENT,
        owner_id INTEGER,
        name TEXT,
        age INTEGER DEFAULT 18,
        position TEXT,
        is_starter INTEGER DEFAULT 0,
        stat_shooting INTEGER,
        stat_dribbling INTEGER,
        stat_keeping INTEGER,
        stat_crossing INTEGER,
        stat_passing INTEGER,
        stat_aerial INTEGER,
        stat_technique INTEGER,
        stat_tackling INTEGER,
        FOREIGN KEY(owner_id) REFERENCES clubs(user_id)
    )''')
    
    conn.commit()
    return conn, c

conn, c = setup_db()

# --- PLAYER GENERATION LOGIC ---

def generate_player_stats(position, min_ovr=50, max_ovr=60):
    """
    Generates stats based on position. 
    LW/ST gets higher attacking stats. GK gets higher keeping.
    """
    stats = {
        "shooting": random.randint(10, 40),
        "dribbling": random.randint(10, 40),
        "keeping": random.randint(5, 15),
        "crossing": random.randint(10, 40),
        "passing": random.randint(10, 40),
        "aerial": random.randint(10, 40),
        "technique": random.randint(10, 40),
        "tackling": random.randint(10, 40)
    }

    # Weight the stats based on position to reach the target 50-60 range
    if position in ["LW", "ST", "RW"]:
        stats["shooting"] = random.randint(min_ovr, max_ovr)
        stats["dribbling"] = random.randint(min_ovr, max_ovr)
    elif position == "GK":
        stats["keeping"] = random.randint(min_ovr, max_ovr)
    elif position in ["CB", "LB", "RB"]:
        stats["tackling"] = random.randint(min_ovr, max_ovr)
        stats["aerial"] = random.randint(min_ovr, max_ovr)
    
    return stats

def calculate_overall(pos, stats):
    """
    FIFA-style Weighted Rating.
    Each position values different stats more.
    """
    s = stats
    if pos in ["ST", "LW", "RW"]:
        ovr = (s['shooting'] * 0.4) + (s['dribbling'] * 0.3) + (s['technique'] * 0.2) + (s['passing'] * 0.1)
    elif pos == "GK":
        ovr = (s['keeping'] * 0.8) + (s['passing'] * 0.2)
    elif pos in ["CB", "LB", "RB"]:
        ovr = (s['tackling'] * 0.4) + (s['aerial'] * 0.3) + (s['passing'] * 0.2) + (s['technique'] * 0.1)
    else: # Midfielders
        ovr = (s['passing'] * 0.4) + (s['technique'] * 0.3) + (s['dribbling'] * 0.2) + (s['crossing'] * 0.1)
    
    return round(ovr)

@bot.tree.command(name="set_starter", description="Move a player to the Starting XI or to the Bench")
@app_commands.describe(player_id="The ID of the player", status="1 for Starter, 0 for Bench")
async def set_starter(interaction: discord.Interaction, player_id: int, status: int):
    user_id = interaction.user.id
    
    if status not in [0, 1]:
        return await interaction.response.send_message("Use `1` to start the player or `0` to bench them.", ephemeral=True)

    # Verify ownership
    c.execute("SELECT name FROM players WHERE player_id = ? AND owner_id = ?", (player_id, user_id))
    player = c.fetchone()
    
    if not player:
        return await interaction.response.send_message("Player not found or you don't own them!", ephemeral=True)

    # If setting as starter, check if they already have 11 starters
    if status == 1:
        c.execute("SELECT COUNT(*) FROM players WHERE owner_id = ? AND is_starter = 1", (user_id,))
        count = c.fetchone()[0]
        if count >= 11:
            return await interaction.response.send_message("You already have 11 starters! Bench someone first.", ephemeral=True)

    c.execute("UPDATE players SET is_starter = ? WHERE player_id = ?", (status, player_id))
    conn.commit()
    
    action = "promoted to the Starting XI" if status == 1 else "moved to the Bench"
    await interaction.response.send_message(f"✅ **{player[0]}** has been {action}.")

# --- HELPER: CALCULATE UPGRADE COST ---
def get_upgrade_cost(current_lvl):
    # Cost increases exponentially: Level 3 to 4 is cheaper than 19 to 20
    return (current_lvl ** 2) * 25000 

# --- COMMAND: UPGRADE FACILITY ---
@bot.tree.command(name="upgrade_training", description="Upgrade your training center (Max Lvl 20)")
async def upgrade_training(interaction: discord.Interaction):
    user_id = interaction.user.id
    
    c.execute("SELECT training_lvl, balance FROM clubs WHERE user_id = ?", (user_id,))
    club = c.fetchone()
    
    if not club:
        return await interaction.response.send_message("Register your club first!", ephemeral=True)
    
    current_lvl, balance = club
    
    if current_lvl >= 20:
        return await interaction.response.send_message("Your Training Center is already at the maximum level (20)!", ephemeral=True)
    
    cost = get_upgrade_cost(current_lvl)
    
    if balance < cost:
        return await interaction.response.send_message(f"Not enough money! Upgrading to Level {current_lvl + 1} costs ${cost:,}.", ephemeral=True)
    
    # Process Upgrade
    new_lvl = current_lvl + 1
    c.execute("UPDATE clubs SET balance = balance - ?, training_lvl = ? WHERE user_id = ?", (cost, new_lvl, user_id))
    conn.commit()
    
    await interaction.response.send_message(f"🏗️ **Upgrade Complete!** Your Training Center is now **Level {new_lvl}**. Next upgrade: ${get_upgrade_cost(new_lvl):,}")

async def aging_loop():

# --- HELPER: CALCULATE TEAM STRENGTHS ---
def calculate_team_power(user_id):
    """Calculates Attack, Midfield, and Defense scores for the starting 11."""
    c.execute("""SELECT position, stat_shooting, stat_passing, stat_tackling, 
                        stat_keeping, stat_technique, stat_dribbling 
                 FROM players WHERE owner_id = ? AND is_starter = 1""", (user_id,))
    starters = c.fetchall()
    
    if len(starters) < 11:
        return None # Cannot play without 11 players

    atk = mid = dfn = 0
    for p in starters:
        pos = p[0]
        # Attack Power: Focus on Shooting & Dribbling
        atk += (p[1] * 0.6) + (p[5] * 0.4)
        # Midfield Power: Focus on Passing & Technique
        mid += (p[2] * 0.5) + (p[5] * 0.5)
        # Defense Power: Focus on Tackling & Keeping
        dfn += (p[3] * 0.7) + (p[4] * 0.3)
            
    return {"atk": atk/11, "mid": mid/11, "dfn": dfn/11}

# --- THE MATCH COMMAND ---
@bot.tree.command(name="challenge", description="Challenge another member to a match!")
async def challenge(interaction: discord.Interaction, opponent: discord.Member):
    if opponent.id == interaction.user.id:
        return await interaction.response.send_message("You can't play against yourself!", ephemeral=True)

    # 1. Get Power for both teams
    team_a = calculate_team_power(interaction.user.id)
    team_b = calculate_team_power(opponent.id)
    
    if not team_a or not team_b:
        return await interaction.response.send_message("One of the teams doesn't have 11 starters!", ephemeral=True)

    # 2. Match Simulation Logic
    score_a = 0
    score_b = 0
    match_events = []

    # Simulate 5 "Chances" per game
    for i in range(5):
        # Phase 1: Who wins the midfield?
        # We add a random factor (1-20) to make it unpredictable
        roll_a = team_a['mid'] + random.randint(1, 20)
        roll_b = team_b['mid'] + random.randint(1, 20)

        if roll_a > roll_b:
            # Team A attacks! Can they beat Team B's defense?
            if (team_a['atk'] + random.randint(1, 20)) > (team_b['dfn'] + random.randint(1, 20)):
                score_a += 1
                match_events.append(f"⚽ **Goal!** {interaction.user.display_name} scores after a brilliant play!")
            else:
                match_events.append(f"🧤 **Save!** {opponent.display_name}'s defense holds firm.")
        else:
            # Team B attacks!
            if (team_b['atk'] + random.randint(1, 20)) > (team_a['dfn'] + random.randint(1, 20)):
                score_b += 1
                match_events.append(f"⚽ **Goal!** {opponent.display_name} hits the back of the net!")
            else:
                match_events.append(f"🛡️ **Block!** {interaction.user.display_name} clears the ball.")

    # 3. Process Results & Rewards
    winner_msg = "It's a Draw!"
    reward = 5000 # Consolation prize
    
    if score_a > score_b:
        winner_msg = f"🏆 {interaction.user.display_name} WINS!"
        reward = 50000
        c.execute("UPDATE clubs SET balance = balance + ? WHERE user_id = ?", (reward, interaction.user.id))
    elif score_b > score_a:
        winner_msg = f"🏆 {opponent.display_name} WINS!"
        reward = 50000
        c.execute("UPDATE clubs SET balance = balance + ? WHERE user_id = ?", (reward, opponent.id))
    
    conn.commit()

    # 4. Display Match Report
    embed = discord.Embed(title="🏟️ Match Results", description="\n".join(match_events), color=0xe74c3c)

@app_commands.describe(player_id="The unique ID of the player you want to buy")
async def buy_player(interaction: discord.Interaction, player_id: int):
    user_id = interaction.user.id
    
    # Check if the player is actually on the market (owner_id = 0)
    c.execute("SELECT name, position, owner_id FROM players WHERE player_id = ?", (player_id,))
    player = c.fetchone()
    
    if not player:
        return await interaction.response.send_message("That player does not exist!", ephemeral=True)
    if player[2] != 0:
        return await interaction.response.send_message("This player is already owned by a club!", ephemeral=True)

    # Check buyer's balance
    c.execute("SELECT balance FROM clubs WHERE user_id = ?", (user_id,))
    club = c.fetchone()
    if not club:
        return await interaction.response.send_message("You need to start a club first!", ephemeral=True)
    
    # Logic: Let's set a flat fee for market players or calculate based on rating
    # For now, let's say a flat $250,000 for free agents
    cost = 250000
    
    if club[0] < cost:
        return await interaction.response.send_message(f"You need ${cost:,} to buy this player!", ephemeral=True)

    # Process Transaction
    c.execute("UPDATE clubs SET balance = balance - ? WHERE user_id = ?", (cost, user_id))
    c.execute("UPDATE players SET owner_id = ?, is_starter = 0 WHERE player_id = ?", (user_id, player_id))
    conn.commit()
    
    embed = discord.Embed(title="🤝 Transfer Complete!", color=0x9b59b6)
    embed.description = f"**{player[0]}** ({player[1]}) has joined your squad for **${cost:,}**!"
    await interaction.response.send_message(embed=embed)

    team_a = calculate_team_power(user_id)
    if not team_a:
        return await interaction.response.send_message("You need 11 starters to play!", ephemeral=True)

    bot_team = {"atk": 55.0, "mid": 55.0, "dfn": 55.0}
    
    score_user = 0
    score_bot = 0
    match_events = []

    # 5 Attack Rounds
    for i in range(5):
        # Midfield Roll
        if (team_a['mid'] + random.randint(-5, 15)) > (bot_team['mid'] + random.randint(-5, 15)):
            # User Attacks
            if (team_a['atk'] + random.randint(-5, 15)) > (bot_team['dfn'] + random.randint(-5, 15)):
                score_user += 1
                match_events.append("🥅 **GOAL!** Your striker found a gap in the bot's defense!")
            else:
                match_events.append("❌ Your attack was intercepted by the Bot defense.")
        else:
            # Bot Attacks
            if (bot_team['atk'] + random.randint(-5, 15)) > (team_a['dfn'] + random.randint(-5, 15)):
                score_bot += 1
                match_events.append("⚠️ **Goal conceded!** The Bot team scored on the counter.")
            else:
                match_events.append("🛡️ Your defenders successfully blocked the Bot.")

    # 3. Rewards
    reward = 0
    if score_user > score_bot:
        reward = 25000 # Lower than PvP but still good
        result_text = "WIN!"
        c.execute("UPDATE clubs SET balance = balance + ? WHERE user_id = ?", (reward, user_id))
    elif score_user == score_bot:
        reward = 5000
        result_text = "DRAW"
        c.execute("UPDATE clubs SET balance = balance + ? WHERE user_id = ?", (reward, user_id))
    else:
        result_text = "LOSS"
    
    conn.commit()

    embed = discord.Embed(title="🎮 Practice Match Result", color=0x34495e)
    embed.add_field(name="Score", value=f"You **{score_user} - {score_bot}** Bot FC", inline=False)
    embed.add_field(name="Result", value=f"{result_text} | Earned: ${reward:,}", inline=True)

# --- 2. GLOBAL ERROR HANDLING ---
# This prevents the bot from "silent failing" when an error occurs
@bot.tree.error
async def on_app_command_error(interaction: discord.Interaction, error: discord.app_commands.AppCommandError):
    if isinstance(error, discord.app_commands.CommandOnCooldown):
        await interaction.response.send_message(f"⏳ Training takes time! Try again in {error.retry_after:.1f}s.", ephemeral=True)
    elif isinstance(error, discord.app_commands.CheckFailure):
        await interaction.response.send_message("❌ You don't have permission to do that.", ephemeral=True)
    else:
        # Log the full error to the console for debugging
        print(f"Error in command: {error}")
        traceback.print_exc()
        
        # Friendly message for the user
        if not interaction.response.is_done():
            await interaction.response.send_message("⚠️ An unexpected error occurred. Please try again later.", ephemeral=True)
