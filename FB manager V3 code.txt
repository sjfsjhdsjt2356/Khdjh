# --- DATABASE INITIALIZATION ---

def setup_db():
    """Initializes the SQLite database and creates all necessary tables."""
    conn = sqlite3.connect('football_rpg.db')
    c = conn.cursor()

    # Table for Club information and finances
    c.execute('''CREATE TABLE IF NOT EXISTS clubs (
        user_id INTEGER PRIMARY KEY,
        club_name TEXT,
        stadium_name TEXT,
        balance INTEGER DEFAULT 1000000,
        training_lvl INTEGER DEFAULT 3,
        formation TEXT DEFAULT '4-4-2'
    )''')

    # Table for Players (Core RPG Data)
    # status codes: 0=Reserve, 1=Starter, 2=Bench, 3=Transfer Listed
    c.execute('''CREATE TABLE IF NOT EXISTS players (
        player_id INTEGER PRIMARY KEY AUTOINCREMENT,
        owner_id INTEGER,
        name TEXT,
        age INTEGER DEFAULT 18,
        position TEXT,
        status INTEGER DEFAULT 0, 
        injury_days INTEGER DEFAULT 0,
        training_until INTEGER DEFAULT 0,
        cooldown_until INTEGER DEFAULT 0,
        stat_shooting INTEGER,
        stat_dribbling INTEGER,
        stat_keeping INTEGER,
        stat_crossing INTEGER,
        stat_passing INTEGER,
        stat_aerial INTEGER,
        stat_technique INTEGER,
        stat_tackling INTEGER,
        FOREIGN KEY(owner_id) REFERENCES clubs(user_id)
    )''')

    # Table for League Standings
    c.execute('''CREATE TABLE IF NOT EXISTS leagues (
        guild_id INTEGER,
        user_id INTEGER,
        points INTEGER DEFAULT 0,
        played INTEGER DEFAULT 0,
        wins INTEGER DEFAULT 0,
        draws INTEGER DEFAULT 0,
        losses INTEGER DEFAULT 0,
        goals_for INTEGER DEFAULT 0,
        goals_against INTEGER DEFAULT 0,
        PRIMARY KEY (guild_id, user_id)
    )''')

    conn.commit()
    return conn, c

# Initialize connection and cursor global variables
conn, c = setup_db()

def talent_suffix():
    """Generates a random suffix for player names to add variety."""
    return random.choice(["Jr.", "Sr.", "II", "III", "IV", "V", ""])

def calculate_ovr(p):
    """
    Calculates the Overall (OVR) rating based on position-specific stats.
    Database indices for stats: 
    9:shooting, 10:dribbling, 11:keeping, 12:crossing, 
    13:passing, 14:aerial, 15:technique, 16:tackling
    """
    pos = p[4]
    try:
        if pos == 'GK':
            # Goalkeepers rely heavily on Keeping and Technique
            return int((p[11] * 0.8) + (p[15] * 0.2))
        elif pos in ['CB', 'LB', 'RB']:
            # Defenders rely on Tackling, Aerial, and Passing
            return int((p[16] * 0.6) + (p[14] * 0.2) + (p[13] * 0.2))
        elif pos in ['CM', 'LM', 'RM']:
            # Midfielders rely on Passing, Technique, and Dribbling
            return int((p[13] * 0.4) + (p[15] * 0.3) + (p[10] * 0.3))
        elif pos in ['ST', 'LW', 'RW']:
            # Attackers rely on Shooting, Dribbling, and Technique
            return int((p[9] * 0.6) + (p[10] * 0.2) + (p[15] * 0.2))
    except (IndexError, TypeError):
        return 50 # Default if data is missing
    return 50

def process_retirement(p_id, p_name, p_age):
    """
    Determines if a player retires based on their age.
    Called inside the daily_maintenance loop (Segment 8).
    """
    if p_age >= 34:
        # Probability of retirement increases with age
        # 34: 20%, 35: 40%, 36: 60%, 37: 80%, 38: 100%
        retirement_chance = (p_age - 33) * 0.2
        if random.random() < retirement_chance:
            c.execute("DELETE FROM players WHERE player_id = ?", (p_id,))
            print(f"👟 RETIREMENT: {p_name} has retired at age {p_age}.")
            return True
    return False
def get_player_price(player_id):
    """
    Calculates a dynamic price based on the player's OVR.
    Base price for 70 OVR is ~$250k. 80+ OVR can cost $1M+.
    """
    c.execute("SELECT * FROM players WHERE player_id = ?", (player_id,))
    p = c.fetchone()
    if not p: return 250000
    
    ovr = calculate_ovr(p)
    
    if ovr >= 80:
        # High-tier players cost significantly more
        return int(500000 + ((ovr - 79) * 150000))
    elif ovr >= 70:
        # Standard mid-tier pricing
        return int(200000 + ((ovr - 69) * 25000))
    else:
        # Budget players
        return 150000

        label='Stadium Name', 
        placeholder='e.g., Old Trafford',
        min_length=3,
        max_length=30

    async def on_submit(self, interaction: discord.Interaction):
        user_id = interaction.user.id

        c.execute("SELECT user_id FROM clubs WHERE user_id = ?", (user_id,))

            return await interaction.response.send_message(
                "❌ You already own a club! Manage it using `/lineup`.", 
                ephemeral=True
            )

        # 1. Insert the new club into the database (Starting balance: $1M)
        c.execute(
            "INSERT INTO clubs (user_id, club_name, stadium_name, balance) VALUES (?, ?, ?, ?)",

        )
        
        # 2. Generate a balanced starting squad of 20 players
        # Guarantee 2 Goalkeepers
        for _ in range(2):
            generate_player(user_id, position='GK')
        
        # Generate 18 more players of random positions

            generate_player(user_id)

        conn.commit()

            title="🎉 Club Established!", 
            description=f"Congratulations, **{interaction.user.name}**!",
            color=discord.Color.green()
        )
        embed.add_field(name="Club Name", value=self.club_n.value, inline=True)

@bot.tree.command(name="train", description="Train a player (5m session, 1h rest)")
@app_commands.describe(player_id="The ID of the player to train")
async def train(interaction: discord.Interaction, player_id: int):
    user_id = interaction.user.id
    now = int(time.time())

    # 1. Fetch Club and Training Level
    c.execute("SELECT training_lvl FROM clubs WHERE user_id = ?", (user_id,))
    club = c.fetchone()
    if not club:
        return await interaction.response.send_message("❌ You don't have a club! Use `/startmyclub`.", ephemeral=True)

    train_lvl = club[0]
    
    # 2. Determine Training Slots (Level 1-5: 1, Level 6-10: 3, Level 11+: 5)
    max_slots = 1 if train_lvl <= 5 else (3 if train_lvl <= 10 else 5)
    
    # 3. Check current active training sessions
    c.execute("SELECT COUNT(*) FROM players WHERE owner_id = ? AND training_until > ?", (user_id, now))
    active_count = c.fetchone()[0]
    
    if active_count >= max_slots:
        return await interaction.response.send_message(
            f"❌ Your Training Center is at capacity ({active_count}/{max_slots} slots). Wait for a session to finish!", 
            ephemeral=True
        )

    # 4. Fetch Player Data
    c.execute("SELECT * FROM players WHERE player_id = ? AND owner_id = ?", (player_id, user_id))
    p = c.fetchone()
    
    if not p:
        return await interaction.response.send_message("❌ Player not found or doesn't belong to you.", ephemeral=True)

    # Database indices: name(2), age(3), training_until(7), cooldown_until(8)
    if p[7] > now:
        rem = (p[7] - now) // 60
        return await interaction.response.send_message(f"🏃 **{p[2]}** is still on the pitch! ({rem}m left)", ephemeral=True)
    
    if p[8] > now:
        rem = (p[8] - now) // 60
        return await interaction.response.send_message(f"😴 **{p[2]}** is exhausted. They need {rem}m more rest.", ephemeral=True)

    # 5. Determine Stat Gain (Buffs for youth)
    # Young players (<=23) get +2, older players get +1
    stat_gain = 2 if p[3] <= 23 else 1
    
    # Randomly select a stat to improve (using your original stat columns)
    stat_options = ["stat_shooting", "stat_dribbling", "stat_crossing", "stat_passing", "stat_aerial", "stat_technique", "stat_tackling"]
    if p[4] == 'GK': stat_options.append("stat_keeping")
    
    chosen_stat = random.choice(stat_options)
    
    # 6. Update Timestamps and Stats
    t_end = now + 300      # 5 Minutes (300 seconds)
    c_end = t_end + 3600   # 1 Hour rest (3600 seconds) after training ends
    
    c.execute(f"UPDATE players SET {chosen_stat} = {chosen_stat} + ?, training_until = ?, cooldown_until = ? WHERE player_id = ?",
              (stat_gain, t_end, c_end, player_id))
    conn.commit()

    await interaction.response.send_message(
        f"🏋️ **{p[2]}** has started an intensive session for **{chosen_stat.replace('stat_', '')}**!\n"
        f"⏱️ Training finishes in 5 minutes.\n"
        f"📈 Gain: +{stat_gain} stat points (Level {train_lvl} Academy)."
    )

@bot.tree.command(name="set_role", description="Assign a player: 1=Starter, 2=Bench, 0=Reserve")
@app_commands.describe(player_id="The ID of the player", role="Role number")
async def set_role(interaction: discord.Interaction, player_id: int, role: int):
    """Moves players between starters, bench, and reserves."""
    user_id = interaction.user.id
    
    if role not in [0, 1, 2]:
        return await interaction.response.send_message("❌ Role must be 1 (Starter), 2 (Bench), or 0 (Reserve).", ephemeral=True)

    # Ownership check
    c.execute("SELECT name, status FROM players WHERE player_id = ? AND owner_id = ?", (player_id, user_id))
    p = c.fetchone()
    if not p:
        return await interaction.response.send_message("❌ Player not found in your club.", ephemeral=True)

    # Limits check
    if role == 1:
        c.execute("SELECT COUNT(*) FROM players WHERE owner_id = ? AND status = 1", (user_id,))
        if c.fetchone()[0] >= 11:
            return await interaction.response.send_message("❌ You already have 11 starters!", ephemeral=True)
    elif role == 2:
        c.execute("SELECT COUNT(*) FROM players WHERE owner_id = ? AND status = 2", (user_id,))
        if c.fetchone()[0] >= 7:
            return await interaction.response.send_message("❌ Your bench is full (max 7)!", ephemeral=True)

    c.execute("UPDATE players SET status = ? WHERE player_id = ?", (role, player_id))
    conn.commit()
    
    role_name = ["Reserve", "Starter", "Bench player"][role]
    await interaction.response.send_message(f"✅ **{p[0]}** is now a **{role_name}**.")

@bot.tree.command(name="buy", description="Buy a player from the Free Agent market")
@app_commands.describe(player_id="The ID of the player to buy")
async def buy(interaction: discord.Interaction, player_id: int):
    user_id = interaction.user.id
    
    # 1. Check if user has a club
    c.execute("SELECT balance FROM clubs WHERE user_id = ?", (user_id,))
    club = c.fetchone()
    if not club:
        return await interaction.response.send_message("❌ You need a club first!", ephemeral=True)

    # 2. Check if player is a Free Agent
    c.execute("SELECT name FROM players WHERE player_id = ? AND owner_id = 0", (player_id,))
    p = c.fetchone()
    if not p:
        return await interaction.response.send_message("❌ This player is not on the Free Agent market.", ephemeral=True)

    # 3. Check Balance with Dynamic Pricing
    cost = get_player_price(player_id) # Call the logic from Segment 2.1
    if club[0] < cost:
        return await interaction.response.send_message(f"❌ You need ${cost:,} to buy this player. Your balance: ${club[0]:,}", ephemeral=True)

    # 4. Complete Transaction
    c.execute("UPDATE clubs SET balance = balance - ? WHERE user_id = ?", (cost, user_id))
    c.execute("UPDATE players SET owner_id = ?, status = 0 WHERE player_id = ?", (user_id, player_id))
    conn.commit()

    await interaction.response.send_message(f"🖋️ **Transaction Complete!** {p[0]} has joined your squad for **${cost:,}**.")

@tasks.loop(hours=24)
async def daily_maintenance():
    """Runs once a day to handle injuries and ageing."""
    now_date = datetime.now()
    
    # 1. Heal Injuries (Reduces days by 1 every 24h)
    c.execute("UPDATE players SET injury_days = MAX(0, injury_days - 1)")
    
    # 2. Monthly Ageing (On the 1st of every month, everyone ages 1 year)
    if now_date.day == 1:
        print("📅 New Month! Processing ageing and retirement...")
        c.execute("SELECT player_id, name, age FROM players")
        all_players = c.fetchall()
        
        for p_id, p_name, p_age in all_players:
            new_age = p_age + 1
            # Check for retirement (Logic defined in Segment 2)
            if process_retirement(p_id, p_name, new_age):
                continue # Player deleted, skip update
            
            c.execute("UPDATE players SET age = ? WHERE player_id = ?", (new_age, p_id))
            
    # 3. Clean up old Free Agents (Removes non-owned players over age 35)
    c.execute("DELETE FROM players WHERE owner_id = 0 AND age > 35")
    
    conn.commit()
    print(f"✅ Maintenance Complete: {now_date.strftime('%Y-%m-%d %H:%M:%S')}")

@bot.tree.command(name="admin_force_result", description="Admin Only: Manually record a league match score")
@app_commands.describe(winner="The winning user", loser="The losing user", score="Format: 3-1")
@app_commands.checks.has_permissions(manage_guild=True)
async def admin_force_result(interaction: discord.Interaction, winner: discord.User, loser: discord.User, score: str):
    """Updates the league table with an official result."""
    guild_id = interaction.guild_id
    
    try:
        w_goals, l_goals = map(int, score.split('-'))
    except:
        return await interaction.response.send_message("❌ Invalid format. Use '3-1'.", ephemeral=True)

    # 1. Update Winner Stats
    c.execute('''UPDATE leagues SET 
                 points = points + 3, played = played + 1, wins = wins + 1, 
                 goals_for = goals_for + ?, goals_against = goals_against + ? 
                 WHERE guild_id = ? AND user_id = ?''', (w_goals, l_goals, guild_id, winner.id))
    
    # 2. Update Loser Stats
    c.execute('''UPDATE leagues SET 
                 played = played + 1, losses = losses + 1, 
                 goals_for = goals_for + ?, goals_against = goals_against + ? 
                 WHERE guild_id = ? AND user_id = ?''', (l_goals, w_goals, guild_id, loser.id))
    
    conn.commit()
    await interaction.response.send_message(f"✅ **Official Result Recorded:** {winner.name} **{score}** {loser.name}")

@bot.tree.command(name="admin_draw", description="Admin Only: Record a league draw")
@app_commands.describe(user1="First player", user2="Second player", score="Format: 1-1")
@app_commands.checks.has_permissions(manage_guild=True)
async def admin_draw(interaction: discord.Interaction, user1: discord.User, user2: discord.User, score: str):
    """Updates the league table for a draw."""
    guild_id = interaction.guild_id
    try:
        goals = int(score.split('-')[0])
    except:
        return await interaction.response.send_message("❌ Invalid format. Use '1-1'.", ephemeral=True)

    # Update both players
    for user in [user1, user2]:
        c.execute('''UPDATE leagues SET 
                     points = points + 1, played = played + 1, draws = draws + 1, 
                     goals_for = goals_for + ?, goals_against = goals_against + ? 
                     WHERE guild_id = ? AND user_id = ?''', (goals, goals, guild_id, user.id))
    
    conn.commit()
    await interaction.response.send_message(f"🤝 **Official Draw Recorded:** {user1.name} and {user2.name} drew **{score}**")
